```javascript
let mic;

function setup() {
  mic = new p5.AudioIn();
  mic.start();  // 브라우저에서 '허용'
}

function draw() {
  let vol = mic.getLevel();
  console.log(vol);
}
```

---

- `sketch.js`

```javascript
let mic;

function setup() {
  createCanvas(200, 200);
  mic = new p5.AudioIn();
  mic.start();  // 브라우저에서 '허용'
}

function draw() {
  background(0);
  let vol = mic.getLevel();
  console.log(vol);
  circle(100, 100, vol*200);
  // ellipse(100, 100, 200, vol*200);
}
```
