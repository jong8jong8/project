- `sketch.js`

```javascript
let x = 200;
let y = 200;

function setup() {
    createCanvas(400, 400);
    // background(0);
}

function draw() {
    // No trails
    background(0);
    x += random(-5, 5);
    y += random(-5, 5);
    fill(255, 200);
    stroke(255);
    rectMode(CENTER);
    rect(x, y, 20, 20);
    
    // Trails
    if (mouseIsPressed) {
        fill(255, 150);
        noStroke();
        ellipse(mouseX, mouseY, 60, 60);
    }

}
```
