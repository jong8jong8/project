- `sketch.js`

```javascript
class Snake {
    constructor() {
        this.x = 0;
        this.y = 0;
        this.xspeed = 1;
        this.yspeed = 0;
        this.total = 0;
        this.tail = [];
    }

    dir(x, y) {
        this.xspeed = x;
        this.yspeed = y;
    }
    
    eat(pos) {
        let d = dist(this.x, this.y, pos.x, pos.y);
        if (d < 1) {
            this.total++;
            return true;
        } else {
            return false;
        }
    }

    death() {
        for (let i = 0; i < this.tail.length; i++) {
            let pos = this.tail[i];
            let d = dist(this.x, this.y, pos.x, pos.y);
            if (d < 1) {
                console.log('starting over');
                startOverSound.play();
                this.total = 0;
                this.tail = [];
            }
        }
    }

    update() {
        if (this.total == this.tail.length) {  // no food eaten
            for (let i = 0; i < this.tail.length - 1; i++) {
                this.tail[i] = this.tail[i + 1];
            }
        }
        this.tail[this.total - 1] = createVector(this.x, this.y);

        this.x = this.x + this.xspeed * scl;
        this.y = this.y + this.yspeed * scl;

        this.x = constrain(this.x, 0, width - scl);
        this.y = constrain(this.y, 0, height - scl);
    }

    show() {
        fill(255);
        for (let i = 0; i < this.tail.length; i++) {
            rect(this.tail[i].x, this.tail[i].y, scl, scl);
        }
        rect(this.x, this.y, scl, scl);
    }
}

let snake;
let scl = 20;
let eatSound;
let startOverSound;
let food;

function pickLocation() {
    let cols = floor(width / scl);  // integer
    let rows = floor(height / scl); // integer
    food = createVector(floor(random(cols)), floor(random(rows)));
    food.mult(scl);
}

function preload() {
    eatSound = loadSound("616696__melokacool__saved.mp3");
    startOverSound = loadSound("322459__lalazzylee1__game-boss-or-character-laughter.mp3");
}

function setup() {
    createCanvas(600, 600);
    snake = new Snake();
    frameRate(10);
    food = createVector(random(width), random(height));
    pickLocation();
}

function draw() {
    background(51);
    snake.death();
    snake.update();
    snake.show();
    if (snake.eat(food)) {
        eatSound.play();
        pickLocation();
    }

    fill(255, 0, 100);
    rect(food.x, food.y, scl, scl);

    
}

function keyPressed() {
    if (keyCode == UP_ARROW) {
        snake.dir(0, -1);
    } else if (keyCode == DOWN_ARROW) {
        snake.dir(0, 1);
    } else if (keyCode == RIGHT_ARROW) {
        snake.dir(1, 0);
    } else if (keyCode == LEFT_ARROW) {
        snake.dir(-1, 0);
    }
}

function mousePressed() {
    snake.total++;
}
