- `sketch.js`

```javascript
let angle = 0;

function setup() {
    createCanvas(400, 300, WEBGL); 
}
  
function draw(){
    background(175);
    rotateX(angle);
    rotateY(angle * 0.3);
    rotateZ(angle * 1.2);
    
    fill(200, 0, 255);
    noStroke();
    torus(100, 10);

    angle += 0.03;
}
```
