- `sketch.js`

```javascript
let angle = 0;
// let kitten;

// function preload() {
//     kitten = loadImage('kitten0.jpg');
// }

function setup() {
    createCanvas(400, 300, WEBGL); 
}
  
function draw(){
    background(0);
    ambientLight(100);
    // ambientMaterial(255);
    normalMaterial();
    directionalLight(255, 255, 255, 0, 0, -1);
    rotateX(angle);
    rotateY(angle * 1.3);
    rotateZ(angle * 0.7);

    // texture(kitten);
    // box(100);

    beginShape();
        vertex(0, 0);
        vertex(100, 0);
        vertex(100, 100);
        vertex(0, 100);
    endShape(CLOSE);
    
    angle += 0.03;
}
```

- `sketch.js`

```javascript
let angle = 0;
let kitten;

function preload() {
    kitten = loadImage('kitten0.jpg');
}

function setup() {
    createCanvas(400, 300, WEBGL); 
}
  
function draw(){
    background(0);
    ambientLight(100);
    ambientMaterial(255);
    normalMaterial();
    directionalLight(255, 255, 255, 0, 0, -1);
    rotateX(angle);
    rotateY(angle * 1.3);
    rotateZ(angle * 0.7);

    // fill(255);
    textureMode(NORMAL);
    texture(kitten);
    translate(-50, -50);
    beginShape();
        vertex(0, 0, 0, 0, 0);
        vertex(100, 0, 0, 1, 0);
        vertex(50, 50, 0, 1, 1);
        vertex(0, 100, 0, 0, 1);
    endShape(CLOSE);
    
    angle += 0.03;
}
```
