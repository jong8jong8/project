---

- `sketch.js`

```javascript
let song;

function preload() {
    song = loadSound("The Columbians - Just Like A Rainbow.mp3");
}

function setup() {
    createCanvas(200, 200);    
}

function draw() {
    background(0);
}

function mousePressed() {
    song.play();
}
```
