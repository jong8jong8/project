- `sketch.js`

```javascript
class Bubble {
    constructor(x, y, r=50) {
        this.x = x;
        this.y = y;
        this.r = r;
        this.brightness = 0;
    }

    move() {
        this.x = this.x + random(-2, 2);
        this.y = this.y + random(-2, 2);
    }

    show() {
        stroke(255);
        strokeWeight(4);
        fill(this.brightness, 125);
        circle(this.x, this.y, this.r*2);
    }

    changeColor(bright) {
        this.brightness = bright;
    }

    contains(x, y) {
        let d = dist(x, y, this.x, this.y);
        if (d < this.r) {
            return true;
        } else {
            return false;
        }
    }
}


let bubble1;
let bubble2;


function setup() {
    createCanvas(600, 400);
    bubble1 = new Bubble(200, 200);
    bubble2 = new Bubble(400, 200, 100);
}


function draw() {
    background(0);
    bubble1.show();
    bubble1.move();
    bubble2.show();
    bubble2.move();
}
```
