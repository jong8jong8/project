- `sketch.js`

```javascript
function setup() {
    createCanvas(600, 400);
}

function draw() {
    background(0);
        
    stroke(255);
    strokeWeight(4);
    noFill();

    let circleX = 300;
    let circleY = 200;
    let circleR = 50;
    let distance = sqrt(
        (mouseX - circleX)*(mouseX - circleX) 
        + (mouseY - circleY)*(mouseY - circleY)
    );
    
    print(distance);

    if (distance < circleR) {
        fill(255, 0, 200); 
    }

    circle(circleX, circleY, circleR*2);   
}
```

```javascript
let x = 0;
let speed = 3;

function setup() {
    createCanvas(600, 400);
}

function draw() {
    background(0);
        
    stroke(255);
    strokeWeight(4);
    noFill();
    circle(x, 200, 100); 

    if (x > width) {
        speed = -3;
    } else if (x < 0) {
        speed = 3;
    }
    x = x + speed;
}
```
