- `sketch.js`

```javascript
let angle = 0;
let kitten;

function preload() {
    kitten = loadImage('kitten0.jpg');
}

function setup() {
    createCanvas(400, 300, WEBGL); 
}
  
function draw(){
    background(175);

    rotateX(angle);
    rotateY(angle * 0.3);
    rotateZ(angle * 1.2);

    texture(kitten);
    noStroke();
    plane(200, 200);
    // box(100);
    // sphere(100);
    // torus(100, 25);

    angle += 0.03;
}
```
