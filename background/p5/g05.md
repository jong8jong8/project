- `sketch.js`

```javascript
let angle = 0;
let sword;

function preload() {
    sword = loadModel('sword.obj');
}
function setup() {
    createCanvas(400, 300, WEBGL); 
}
  
function draw(){
    background(175);
    
    ambientLight(100);
    ambientMaterial(255);
    directionalLight(255, 255, 255, 0, 0, -1);
    rotateX(angle);
    rotateY(angle * 1.3);
    rotateZ(angle * 0.7);

    // box(100);

    scale(4);
    model(sword);
    
    angle += 0.03;
}
```
